## logFC comparison

### logFC on original 
```{r limma}
## original devs
colData(dev)$group_id <- substr(colnames(dev), 1, nchar(colnames(dev)) - 16)
design <- model.matrix(~ dev$group_id)
fit <- eBayes(lmFit(assays(dev)$z, design))
res <- topTable(fit, n = Inf)
ids <- match(rownames(res), rownames(dev))
res$name <- rowData(dev)$name[ids]
res$rank <- rank(res$P.Value)

## TMM
colData(TMM_devs)$group_id <- substr(colnames(TMM_devs), 1, nchar(colnames(TMM_devs)) - 16)
design <- model.matrix(~ TMM_devs$group_id)
fit <- eBayes(lmFit(assays(TMM_devs)$z, design))
TMMres <- topTable(fit, n = Inf)
ids <- match(rownames(TMMres), rownames(TMM_devs))
TMMres$name <- rowData(TMM_devs)$name[ids]
TMMres$rank <- rank(TMMres$P.Value)

```

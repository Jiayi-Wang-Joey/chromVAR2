applyLimma <- \(devs, assay.id = "z") {
  res <- lapply(devs, \(dev) {
    colData(dev)$group_id <- substr(colnames(dev), 1, nchar(colnames(dev)) - 5)
    design <- model.matrix(~ dev$group_id)
    fit <- eBayes(lmFit(assay(dev, assay.id), design))
    res <- topTable(fit, n = Inf)
    ids <- match(rownames(res), rownames(dev))
    res$name <- rowData(dev)$name[ids]
    res 
    })
  names(res) <-  c("0", "0.25", "0.5", "1", "3")
  res
}
limma_thr25 <- applyLimma(devs_thr25) 
limma_thr50 <- applyLimma(devs_thr50)
limma_thr75 <- applyLimma(devs_thr75)
limma_thr100 <- applyLimma(old_devs)

lfc <- lapply(names(limma_thr25), \(x) {
    thr25 <- data.frame(limma_thr25[[x]], effect = x, threshold = "25%")
    thr50 <- data.frame(limma_thr50[[x]], effect = x, threshold = "50%")
    thr75 <- data.frame(limma_thr75[[x]], effect = x, threshold = "75%")
    thr100 <- data.frame(limma_thr100[[x]], effect = x, threshold = "100%")
    do.call(rbind, list(thr25, thr50, thr75, thr100))
})
lfc <- do.call(rbind, lfc)

applyLimma <- \(devs, assay.id = "z") {
  res <- lapply(devs, \(dev) {
    colData(dev)$group_id <- substr(colnames(dev), 1, nchar(colnames(dev)) - 5)
    design <- model.matrix(~ dev$group_id)
    fit <- eBayes(lmFit(assay(dev, assay.id), design))
    res <- topTable(fit, n = Inf)
    ids <- match(rownames(res), rownames(dev))
    res$name <- rowData(dev)$name[ids]
    res 
    })
  names(res) <-  c("0", "0.25", "0.5", "1", "3")
  res
}
limma_thr25 <- applyLimma(devs_thr25) 
limma_thr50 <- applyLimma(devs_thr50)
limma_thr75 <- applyLimma(devs_thr75)
limma_thr100 <- applyLimma(old_devs)

lfc <- lapply(names(limma_thr25), \(x) {
    thr25 <- data.frame(limma_thr25[[x]], effect = x, threshold = "25%")
    thr50 <- data.frame(limma_thr50[[x]], effect = x, threshold = "50%")
    thr75 <- data.frame(limma_thr75[[x]], effect = x, threshold = "75%")
    thr100 <- data.frame(limma_thr100[[x]], effect = x, threshold = "100%")

        annotations = motif_ix,
        background_peaks = bg
        )

Differential analysis of deviations between conditions

Code 
applyLimma <- \(devs, assay.id = "z") {
  res <- lapply(devs, \(dev) {
    colData(dev)$group_id <- substr(colnames(dev), 1, nchar(colnames(dev)) - 5)
    design <- model.matrix(~ dev$group_id)
    fit <- eBayes(lmFit(assay(dev, assay.id), design))
    res <- topTable(fit, n = Inf)
    ids <- match(rownames(res), rownames(dev))
    res$name <- rowData(dev)$name[ids]
    res$rank <- rank(res$P.Value)
    res 
    })
}
old_limma <- applyLimma(old_devs)
csaw_limma <- applyLimma(csaw_devs, assay = "offset")
TMM_limma <- applyLimma(TMM_devs)
names(TMM_limma) <- names(sims)
names(old_limma) <- names(sims)
names(csaw_limma) <- names(sims)
